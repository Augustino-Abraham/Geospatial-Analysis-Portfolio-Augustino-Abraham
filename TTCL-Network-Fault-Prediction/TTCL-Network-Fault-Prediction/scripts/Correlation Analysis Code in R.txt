# ---------------- Build Model ----------------
build_model <- function(data) {
  data$faults$elevation <- terra::extract(data$elevation, vect(data$faults))[,2]
  data$faults$rainfall <- terra::extract(data$rainfall, vect(data$faults))[,2]
  data$faults$soil_moisture <- terra::extract(data$soil_moisture, vect(data$faults))[,2]
  data$faults$tmax <- terra::extract(data$tmax, vect(data$faults))[,2]
  data$faults$tmin <- terra::extract(data$tmin, vect(data$faults))[,2]
  data$faults$building_density <- lengths(st_intersects(st_buffer(data$faults, 50), data$buildings))
  data$faults$fault_present <- 1
  
  set.seed(123)
  bg_points <- st_sample(data$dodoma_urban, 500, type = "random") %>% st_as_sf()
  bg_points <- st_transform(bg_points, 3857)
  bg_points$elevation <- terra::extract(data$elevation, vect(bg_points))[,2]
  bg_points$rainfall <- terra::extract(data$rainfall, vect(bg_points))[,2]
  bg_points$soil_moisture <- terra::extract(data$soil_moisture, vect(bg_points))[,2]
  bg_points$tmax <- terra::extract(data$tmax, vect(bg_points))[,2]
  bg_points$tmin <- terra::extract(data$tmin, vect(bg_points))[,2]
  bg_points$building_density <- lengths(st_intersects(st_buffer(bg_points, 50), data$buildings))
  bg_points$fault_present <- 0
  
  combined_data <- bind_rows(st_drop_geometry(data$faults), st_drop_geometry(bg_points))
  
  model <- glm(fault_present ~ elevation + rainfall + soil_moisture + tmax + tmin + building_density,
               data = combined_data, family = binomial())
  predicted_probs <- predict(model, type = "response")
  roc_obj <- roc(combined_data$fault_present, predicted_probs)
  
  list(model = model, roc = roc_obj, combined_data = combined_data)
}

# --------------- Load model data ----------------
model_data <- build_model(load_data())

# ---------------- Select variables for a broader pairwise analysis ----------------
correlation_data <- model_data$combined_data %>%
  select(elevation, rainfall, soil_moisture, tmax, tmin, fault_present) %>%
  mutate(fault_present = as.factor(fault_present))  # ensure it's a factor

# --------------- Visualize pairwise relationships with color indicating fault presence ----------------
ggpairs(correlation_data, aes(color = fault_present),
        title = "Pairwise Correlation & Distribution: Faults vs Variables")
